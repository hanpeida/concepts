\begin{problem}

  Suppose $f(x) = \sqrt{x+1} - \sqrt{x}$.  What is true about $f$?
  \begin{multipleChoice}
    \choice{When $x$ is very large, $f(x)$ is very large.}
    \choice[correct]{When $x$ is very large, $f(x)$ is close to $0$.}
    \choice{When $x$ is very large, $f(x)$ is close to $1$.}
    \choice{When $x$ is a positive number close to $0$, $f(x)$ is close to $1$.}
    \choice{When $x$ is close to one, $f(x)$ is close to $0$.}
  \end{multipleChoice}
\end{problem}
