\begin{document}
\begin{problem}

  Let $x = \sqrt{1001} - \sqrt{1000}$ and $y = \sqrt{101} - \sqrt{100}$.  How do $x$ and $y$ compare?
  \begin{multipleChoice}
    \choice[correct]{$x < y$.}
    \choice{$x = y = 1$.}
    \choice{$x = y \neq 1$.}
    \choice{$x > y$.}
  \end{multipleChoice}
\end{problem}
