\begin{problem}

  Let $f$ be a polynomial function.  Assume you know that the rate of
  change of $f$ is always increasing on the $[a,b]$.  What can you say
  about $f$?
  \begin{multipleChoice}
    \choice{$f$ must have an absolute maximum value on $[a,b]$ but it is unknown whether it has an absolute minimum value.}
    \choice{$f$ must have an absolute minimum value on $[a,b]$ but it is unknown whether it has an absolute maximum value.}
    \choice[correct]{$f$ must have both an absolute maximum value and an absolute minimum value.}
    \choice{It is not guaranteed that $f$ will have an absolute maximum or an absolute minimum value on $[a,b]$.}
  \end{multipleChoice}
\end{problem}
