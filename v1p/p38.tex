\begin{problem}

  A certain quantity $x$ depends on time $t$.  The rate of change of
  $x$ with respect to $t$ is increasing.  What can you conclude?
  \begin{multipleChoice}
    \choice{The quantity $x$ is increasing.}
    \choice{$x$ may not be increasing now, but if I wait long enough, eventually $x$ will be increasing.}
    \choice{The quantity $x$ is decreasing.}
    \choice{$x$ may not be decreasing now, but if I wait long enough, eventually $x$ will be decreasing.}
    \choice[correct]{We cannot determine whether $x$ is increasing or decreasing.}
  \end{multipleChoice}
\end{problem}
