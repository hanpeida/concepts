\begin{problem}

  Suppose $f$ is a quotient of polynomials, and $f(0) = 0$.  When $h$
  is a number close to zero, what do you expect $\frac{f(h)}{h}$ is
  closest to?
  \begin{multipleChoice}
    \choice{$f(0)$.}
    \choice{$h$.}
    \choice{$f(h)$.}
    \choice[correct]{We cannot tell.}
  \end{multipleChoice}
\end{problem}
