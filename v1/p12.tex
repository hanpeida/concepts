\begin{problem}

  %\GoodQuestions{Subject: Mean Value Theorem and shapes of curves 9D}
  Two racers start a race at the same moment and finish in a tie.
  Which of the following must be true?
  \begin{multipleChoice}
    \choice{At some point during the race the two racers were not tied.}
    \choice{The racers' speeds at the end of the race must have been exactly the same.}
    \choice[correct]{The racers must have had the same speed at exactly the same
      time at some point in the race.}
    \choice{The racers had to have the same speed at some moment, but not
      necessarily at exactly the same time.}
  \end{multipleChoice}
\end{problem}
