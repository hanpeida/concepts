\begin{problem}

  Suppose $A$ is just a bit bigger than $a$ and $B$ is just a bit
  bigger than $b$.  In other words, $A > a$ and $B > b$ but
  $A \approx a$ and $B \approx b$ are close to zero.  How does $AB$
  compare to $ab$?
  \begin{multipleChoice}
    \choice{$AB$ is bigger than $ab$.}
    \choice{$AB$ is equal to $ab$.}
    \choice{$AB$ is smaller than $ab$.}
    \choice[correct]{It could be that $AB$ or $ab$ is larger.}
  \end{multipleChoice}
\end{problem}
