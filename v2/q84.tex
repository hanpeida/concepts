\begin{problem}
  Let $x = e^{10001} - e^{10000}$ and $y = e^{10002} - e^{10001}$.  How do $x$ and $y$ compare?
  \begin{multipleChoice}
    \choice[correct]{$x < y$}
    \choice{$x = y$}
    \choice{$x > y$}
  \end{multipleChoice}
\end{problem}
