\begin{problem}
  point $(a,f(a))$, the zoomed-in graph looks like a straight line.
  When $h$ is very small, how does $f(a)$ compare to $f(a+h)$?
  \begin{multipleChoice}
    \choice[correct]{$f(a)$ is approximately $f(a+h)$ (i.e., $f(a) \approx f(a+h)$)}
    \choice{$f(a)$ need not be close to $f(a+h)$ (e.g., $f(x) = 1/x$ but $f(-0.001) \not\approx f(0.001)$)}
  \end{multipleChoice}
\end{problem}
