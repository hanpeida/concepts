\begin{problem}
  Suppose $z = 0.0001 + 10^{-9}$ and $x = \sqrt{z}$ and $y = \sqrt{x}$.  What integer is closest to $y/x$?
  \begin{multipleChoice}
    \choice{$10^9$}
    \choice{$100$}
    \choice[correct]{$10$}
    \choice{$0$}
  \end{multipleChoice}
\end{problem}
