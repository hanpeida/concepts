\begin{problem}
  Suppose $f(x)^2 + g(x)^2 = 10$ and suppose $f'(a) > 0$.  What must be true about $g'(a)$?
  \begin{multipleChoice}
    \choice{$g'(a) > 0$}
    \choice{$g'(a) = 0$}
    \choice{$g'(a) < 0$}
    \choice[correct]{None of the above statements are necessarily true.}
  \end{multipleChoice}
\end{problem}
