\begin{problem}
  Suppose $f$ is a differentiable function and $f(0) = 0$ and
  $f(\sin \frac{\pi}{100}) = 9$.  What number below is the best guess for $f(0.01)$?
  \begin{multipleChoice}
    \choice{$1$}
    \choice[correct]{$3$}
    \choice{$4$}
    \choice{$6$}
    \choice{$9$}
  \end{multipleChoice}
\end{problem}
