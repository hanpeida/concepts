\begin{problem}
	$f$ has a horizontal asymptote of $y = 7$.  Which of the following is true?
	\begin{multipleChoice}
		\choice{The graph of $f$ never crosses the line $y=7$}
		\choice{The graph of $f$ can only cross the line $y=7$ a finite number of times}
		\choice[correct]{The graph of $f$ could cross the line $y=7$ infinitely many times}
	\end{multipleChoice}
\end{problem}
