\begin{problem}
  When $n$ is a large integer, what can be said about $\int_0^{n \pi} \sin x \, dx$.

    \choice{It is large.}
    \choice{It is close to zero but not exactly zero.}
    \choice[correct]{It is exactly zero.}
    \choice{It is very negative.}
  \end{multipleChoice}
\end{problem}
