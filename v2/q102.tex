\begin{problem}
  Suppose $f(x) = e^x$ so that $f'(x) = f(x)$.  Let $g(x) = e^{f(x)}$.  What is $g'(x)$?
  \begin{multipleChoice}
    \choice{$g'(x) = e^{(x e^x)}$ by the chain rule}
    \choice[correct]{$g'(x) = e^{(x + e^x)}$ by the chain rule}
    \choice{$g'(x) = e^{(e^x)}$ by the chain rule}
  \end{multipleChoice}
\end{problem}
