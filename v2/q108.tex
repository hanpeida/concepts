\begin{problem}
  Suppose $f$ is a differentiable function with domain the real numbers; further suppose that $f(0) = 0$ and that $f'(x) < 1$ for all $x$.  What is a possible value for $f(0.9999)$?
  \begin{multipleChoice}
    \choice[correct]{$0.99$}
    \choice{$0.99999$}
    \choice{$1.00$}
    \choice{$1.0001$}
  \end{multipleChoice}
\end{problem}
