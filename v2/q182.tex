\begin{problem}
  Suppose $W(t)$ is the weight of a person at $t$ years of age.  What
  might $\int_{18}^{19} W'(t) \, dt = 15$ mean?
  \begin{multipleChoice}
    \choice[correct]{First year students gain 15 pounds.}
    \choice{First year students lose 15 pounds.}
  \end{multipleChoice}
\end{problem}
