    }
    \draw (0.85,0) node [circle,fill,inner sep=1pt,label=above:$x$](e){};
    \draw (3.40,0) node [circle,fill,inner sep=1pt,label=above:$y$](e){};
  \end{tikzpicture}
  \end{image}
  What is the best guess for $y/x$?
  \begin{multipleChoice}
    \choice{2.89}

    \choice[correct]{4}

  \end{multipleChoice}
