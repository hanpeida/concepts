\begin{problem}
  Pick a large number $x$ and a number $h$ very close to $0$ so that $0 > \sin x > \sin (x+h)$.  How might $\cos (x+h)$ compare to $\cos x$?
  \begin{multipleChoice}
    \choice[correct]{$\cos (x+h) > \cos x$}
    \choice{$\cos (x+h) = \cos x$}
    \choice{$\cos (x+h) < \cos x$}
  \end{multipleChoice}
\end{problem}
