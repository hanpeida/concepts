\begin{problem}
  I begin the game by setting $x$ to be 1~billion.  Each turn, I
  choose to either subtract one from $x$ or to cut $x$ in half.  After
  30 turns, can I possibly end up with a negative number?
  \begin{multipleChoice}
    \choice[correct]{Yes}
    \choice{No}
  \end{multipleChoice}
\end{problem}
