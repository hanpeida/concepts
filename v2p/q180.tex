\begin{problem}
  Suppose the domain of $f$ is all real numbers, and $f'$ exists and
  is continuous, and $f'$ is negative.  What can one say about $f$?
  \begin{multipleChoice}
    \choice[correct]{The function $f$ is one-to-one.}
    \choice{The function $f$ is not necessarily one-to-one.}
  \end{multipleChoice}
\end{problem}
