\begin{problem}
  Pick two numbers $a$ and $b$ between $0$ and $\pi$ so that
  $a \approx b$.  Pick a very large integer $N$.  How does $\sin^N a$
  compare to $\sin^N b$?
  \begin{multipleChoice}
    \choice{$\sin^N a$ is very close to $\sin^N b$ because $a \approx b$}
    \choice[correct]{$\sin^N a$ need not be very close to $\sin^N b$ even though $a \approx b$}
  \end{multipleChoice}
\end{problem}
