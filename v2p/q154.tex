\begin{problem}
  Suppose $f$ is a positive, increasing, continuous function.  Define $F(x) = \int_0^x f(t) \, dt$.  What can be said about $F$?
  \begin{multipleChoice}
    \choice[correct]{$F(2.1) - F(2) > F(2) - F(1.9)$}
    \choice{$F(2.1) - F(2) = F(2) - F(1.9)$}
    \choice{$F(2.1) - F(2) < F(2) - F(1.9)$}
  \end{multipleChoice}
\end{problem}
