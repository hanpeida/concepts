\begin{problem}
  Suppose $p(x) = x^2 + bx + c$ is a quadratic polynomial.  If $c > 0$, is it possible that there is a point $x$ so that $p'(x) = p(x) = 0$?
  \begin{multipleChoice}
    \choice[correct]{Yes}
    \choice{No}
  \end{multipleChoice}
\end{problem}
