\begin{problem}
  Suppose $f$ is a continuous, positive function.  Let $A = \int_0^2 f(x) \, dx$ and $B = \int_0^1 f(2x) \, dx$.  How do $A$ and $B$ compare?
  \begin{multipleChoice}
    \choice[correct]{$A$ is larger}
    \choice{$B$ is larger}
    \choice{$A$ and $B$ are equal}
  \end{multipleChoice}
\end{problem}
