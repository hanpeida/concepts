\begin{problem}
  Suppose $f$ is a differentiable function.  When $h$ is a number
  close to zero, what do you expect $\frac{f(4h)}{2h}$ is close to?
  \begin{multipleChoice}
    \choice{$\frac{1}{4} \, f'(0)$}
    \choice{$\frac{1}{2} \, f'(0)$}
    \choice{$f'(0)$}
    \choice[correct]{$2 \, f'(0)$}
    \choice{$4 \, f'(0)$}
  \end{multipleChoice}
\end{problem}
