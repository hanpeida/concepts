\begin{problem}
  Suppose $f$ is a continuous, positive function so that $f(x) \approx 2$ for all $x$.  When $n$ is very large, about how big is $\int_0^1 f(nx) \, dx$?
  \begin{multipleChoice}
    \choice{About $0$}
    \choice[correct]{About $2$}
    \choice{About $2n$}
    \choice{Much larger than $2n$}
   \end{multipleChoice}
\end{problem}
