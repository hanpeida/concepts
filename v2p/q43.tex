\begin{problem}
  Suppose $x$ is close to $0$ and $y$ is close to $-2$ and $z$ is close to $2$.  What is true about these numbers?
  \begin{multipleChoice}
    \choice{$y/x < z/x$}
    \choice[correct]{$y/x^4 < z/x^2$}
    \choice{$y/x^5 < z/x^3$}
    \choice{$y/x^5 < z/x^5$}
  \end{multipleChoice}
\end{problem}
