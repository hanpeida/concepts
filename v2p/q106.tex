\begin{problem}

  If $f$ and $g$ are both differentiable and $h=f\circ g$,
  $h'(2)$ equals
  \begin{multipleChoice}
    \choice{$f'(2)\circ g'(2)$}
    \choice{$f'(2)g'(2)$}
    \choice[correct]{$f'(g(2)) g'(2)$}
    \choice{$f'(g(x)) g'(2)$}
  \end{multipleChoice}
\end{problem}
