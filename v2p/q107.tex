\begin{problem}

  The area of a circle, $A=\pi r^2$, changes as its radius changes. If
  the radius changes with respect to time, the change in area with
  respect to time is
  \begin{multipleChoice}
    \choice{$\frac{dA}{dr}=2\pi r$}
    \choice{$\frac{dA}{dt}=2\pi r+\frac{dr}{dt}$}
    \choice[correct]{$\frac{dA}{dt}=2\pi r \frac{dr}{dt}$}
    \choice{not enough information}
  \end{multipleChoice}
\end{problem}
