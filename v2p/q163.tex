\begin{problem}
  Suppose $f(a)$ is the area of the region bounded by $0 \leq y \leq x^2$ and $0 \leq x \leq a$.  What integer is closest to $\frac{f(3.01) - f(2.99)}{0.01}$?
  \begin{multipleChoice}
    \choice{3}
    \choice{9}
    \choice{12}
    \choice[correct]{18}
  \end{multipleChoice}
\end{problem}
