\begin{problem}
  Suppose a function $f$ has the property that $f'' = -f$.  What is true of $f$?
  \begin{multipleChoice}
    \choice{$f(x) = \cos x$}
    \choice[correct]{$f^{(4)}(x) = f(x)$}
    \choice{$f(x) = \tan x$}
    \choice{$f^{(4)}(x) = -f(x)$}
  \end{multipleChoice}
\end{problem}
