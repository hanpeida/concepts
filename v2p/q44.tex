\begin{problem}
  Suppose $x$ is $0.000001$ and $y = 10$ and $z = 9$.  Which is larger:
  $A = \frac{\sin (yx)}{zx}$ or $B = \frac{\sin (zx)}{yx}$?
  \begin{multipleChoice}
    \choice[correct]{$A$ is larger than $B$.}
    \choice{$B$ is larger than $A$.}
  \end{multipleChoice}
\end{problem}
